CUSTOMER TABLE

CREATE TABLE CUSTOMER(
TITLE VARCHAR2(10),
FIRST_NAME VARCHAR2(10),
LAST_NAME VARCHAR2(10),
EMAIL VARCHAR2(25),
PASSWORD VARCHAR2(15),
DATE_OF_BIRTH DATE,
PHONENUMBER NUMBER(20),
CONSTRAINT CUSTOMER_PRIMARY_KEY PRIMARY KEY (EMAIL)
);
-----------------------------------------


FLIGHT TABLE

CREATE TABLE FLIGHT(
FLIGHT_NUMBER NUMBER(15),
SOURCE VARCHAR2(15),
DESTINATION VARCHAR2(15),
ARRIVAL_TIME TIMESTAMP,
DEPARTURE_TIME TIMESTAMP,
DURATION TIMESTAMP,
SEATS NUMBER(5),
TRAVEL_DATE DATE,
RETURN_DATE DATE,
CONSTRAINT FLIGHT_PRIMARY_KEY PRIMARY KEY (FLIGHT_NUMBER,TRAVEL_DATE)
);
-------------------------------------


FLIGHT_BOOKED TABLE

CREATE TABLE FLIGHT_BOOKED(
FLIGHT_NUMBER NUMBER(15),
BOOKING_ID NUMBER(15),
TRANSACTION NUMBER(15),
EMAIL VARCHAR2(25),
BOOKING_DATE DATE,
ADMIN_FLAG NUMBER,
CUSTOMER_FLAG NUMBER,
TRAVEL_DATE DATE,
CONSTRAINT FBOOKED_PRIMARY_KEY PRIMARY KEY (BOOKING_ID),
CONSTRAINT FBOOKED_FNO_FOREIGN_KEY FOREIGN KEY (FLIGHT_NUMBER,TRAVEL_DATE) REFERENCES FLIGHT (FLIGHT_NUMBER,TRAVEL_DATE),
CONSTRAINT FBOOKED_EMAIL_FOREIGN_KEY FOREIGN KEY (EMAIL) REFERENCES CUSTOMER (EMAIL)
);
---------------------------

ADMIN TABLE

CREATE TABLE ADMIN(
USERNAME VARCHAR2(20),
PASSWORD VARCHAR2(15),
CONSTRAINT ADMIN_PRIMARY_KEY PRIMARY KEY (USERNAME)
);
